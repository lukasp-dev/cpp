# Multi-threading

C++ 에는 Multithreading 을 구현하는 방법이 여러가지가 있는데 어떤게 있는지 잘 알아보자.

### 1) `std:thread`
- All parameters are passed by **value** unless used `std::ref` to pass by **reference**.
```cpp
#include <iostream>
#include <vector>      // std::vector 사용을 위해 필요
#include <thread>      // std::thread 사용을 위해 필요
#include <numeric>     // std::accumulate 사용을 위해 필요
#include <cstdint>     // uint64_t 정의를 위해 필요 (uint64_t.h는 없음)

// 1. sum을 참조(&)로 받으므로 호출 시 std::ref가 필수입니다.
void AccumulateRange(uint64_t &sum, uint64_t start, uint64_t end) {
    sum = 0;
    for (uint64_t i = start; i < end; ++i) {
        sum += i;
    }
}

int main() {
    const int number_of_threads = 2;
    // 2. 10억은 int 범위를 넘을 수 있으므로 안전하게 uint64_t로 선언합니다.
    const uint64_t number_of_elements = 1000 * 1000 * 1000;
    const uint64_t step = number_of_elements / number_of_threads;
    
    std::vector<uint64_t> partial_sums(number_of_threads);

    // 3. std::thread 생성 및 실행
    // t2의 끝 범위를 number_of_threads * step 대신 number_of_elements로 써도 무방합니다.
    std::thread t1(AccumulateRange, std::ref(partial_sums[0]), 0, step);
    std::thread t2(AccumulateRange, std::ref(partial_sums[1]), step, number_of_elements);

    // 4. 스레드가 끝날 때까지 대기
    t1.join();
    t2.join();

    // 5. 최종 합산
    uint64_t total = std::accumulate(partial_sums.begin(), partial_sums.end(), uint64_t(0));
    std::cout << "total: " << total << std::endl;

    return 0;
}
```

```cpp
#include <iostream>
#include <vector>
#include <thread>
#include<chrono>
#include <numeric>
#include <iomanip>

using namespace std;
const uint64_t operations = 100'000'000;

void Task(uint64_t& sum, uint64_t count) {
    for(uint64_t i=0; i<count; ++i){
        sum++; // 여기서 계속 메모리에 접근하며 경합 발생
    }
}


void RunTest(int num_threads) {
    vector<uint64_t> partial_sums(num_threads, 0);
    vector<thread> threads;
    uint64_t count_per_thread = operations / num_threads;

    auto start = chrono::high_resolution_clock::now();

    for(int i=0; i < num_threads; i++) {
        threads.emplace_back(Task, ref(partial_sums[i]), count_per_thread);
    }

    for(auto& t : threads) {
        t.join();
    }

    auto end = chrono::high_resolution_clock::now();
    chrono::duration<double, milli> elapsed = end - start;

    cout << "Threads: " << setw(3) << num_threads
    << " | Time: " << setw(10) << elapsed.count() << " ms" << endl;
}

int main() {
    // 1개(순차 실행)부터 128개까지 늘려가며 테스트
    vector<int> thread_counts = {1, 2, 4, 8, 16, 32, 64, 128};
    
    cout << "--- Starting Multi-threading Performance Test ---" << endl;
    for (int count : thread_counts) {
        RunTest(count);
    }
    return 0;
}
```

### 2) using a functor

```cpp
#include <iostream>
#include <vector>
#include <thread>
#include <numeric>
#include <functional> // Required for std::ref

class AccumulateFunctor {
public:
    // Fixed typo: changed t_sum to _sum to match the member name
    AccumulateFunctor() : _sum{0} {}

    void operator()(uint64_t start, uint64_t end) {
        _sum = 0; // Local accumulation
        for (uint64_t i = start; i < end; ++i) {
            _sum += i;
        }
    }

    uint64_t _sum; // Moved to public so main() can read it
};

int main() {
    const int number_of_threads = 10;
    uint64_t number_of_elements = 1000'000'000; 
    uint64_t step = number_of_elements / number_of_threads;

    std::vector<std::thread> threads;
    std::vector<AccumulateFunctor*> functors;

    // Launching threads
    for (int i = 0; i < number_of_threads; ++i) {
        AccumulateFunctor* functor = new AccumulateFunctor();
        functors.push_back(functor);
        
        uint64_t start = i * step;
        // Ensures the last thread covers any remainder
        uint64_t end = (i == number_of_threads - 1) ? number_of_elements : (i + 1) * step;

        // Use std::ref to ensure the thread works on the actual object on the heap
        threads.emplace_back(std::ref(*functor), start, end);
    }

    // Joining threads: Wait for all work to finish
    for (std::thread & t : threads) {
        if (t.joinable()) {
            t.join();
        }
    }

    // Aggregatin g results: The final "Reduction" step
    uint64_t total = 0; 
    for (auto pf : functors) {
        total += pf->_sum;
    }

    std::cout << "Total: " << total << std::endl; // 499999999500000000

    // Manual memory management
    for (auto pf : functors) {
        delete pf;
    }

    return 0;
}
```
